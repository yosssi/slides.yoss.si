Hello world!

package main

import "github.com/go-martini/martini"

func main() {
  m := martini.Classic()
  m.Get("/", func() string {
    return "Hello world!"
  })
  m.Run()
}

Basics

  • Routes
  • Handlers

Routes

  • An HTTP method paired with a URL-matching pattern 
    m.Get("/foo", func() {
  // show something
})

m.Post("/bar", func() {
  // create something
})

Handlers

  • Any kind of callable functions 
    m.Get("/foo", func() {
  // show something
})
  • "middleware handler" is a handler invoked before routing 
    m.Use(func() {
  // do some middleware stuff
})

m.Get("/foo", func() {
  // show something
})

Utilizing dependency injection

  • Martini injects structs into arguments of handlers via reflection
    m.Get("/hello/:name", func(params martini.Params) string {
  // params is injected by Martini
  return "Hello " + params["name"]
})
    func SetupDB() *sql.DB {
  db, _ := sql.Open("postgres", "dbname=sample")
  return db
}

// DB handle will be availabe to
// all handlers as *sql.DB
m.Map(SetupDB())

m.Get("/users", func(db *sql.DB) string {
  // db is injected by Martini
  rows, err := db.Query("SELECT id FROM users")
})

Compatible with http package

  • We can use http.ResponseWriter and http.Request in handlers
    m.Get("/", func(res http.ResponseWriter, req *http.Request) {
  // res and req are injected by Martini
  if req.Header.Get("X-API-KEY") != "secret123" {
    res.WriteHeader(http.StatusUnauthorized)
  }
  // ...
})

Compatible with http package

  • A Martini struct implements http.Handler and can be linked up to other servers 
    type Handler interface {
  ServeHTTP(ResponseWriter, *Request)
}
    func (m *Martini) ServeHTTP(
  res http.ResponseWriter,
  req *http.Request,
) {
  m.createContext(res, req).run()
}
    m := martini.Classic()
http.Handle("/", m) // Register the Martini struct as a handler
// ...
http.ListenAndServe(":8080", nil)

build a single binary file

    func public_css_all_min_css() ([]byte, error) {
  return bindata_read([]byte{
    0x1f, 0x8b, 0x08, 0x00, 0x00, 0x09, 0x6e, 0x88,
    0xdd, 0x6a, 0xe3, 0x3c, 0x10, 0x7d, 0x15, 0x43,
    // ...
  })
}
  2. Pass Asset function generated by go-bindata to middlewares
    // Serve static files from binary data
m.Use(staticbin.Static("static", Asset))

// Serve html templates from binary data
m.Use(render.Renderer(render.Options{
  Asset:      Asset,
  AssetNames: AssetNames(),
}))
